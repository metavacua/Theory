\section{Structural Rules}

\begin{center}
	\[
	\begin{prooftree}
	\infer0[Id]{A ⊢ A}
	\end{prooftree}
	\]
	
	\[
	\begin{prooftree}
	\hypo{Γ_{L}, A ⊢}
	\hypo{⊢ Δ_{L}, A}
	\infer2[Cut]{Γ_{L} ⊢ Δ_{L}}
	\end{prooftree}
	\]
	
	\[
	\begin{prooftree}
	\hypo{A, Γ_{R} ⊢}
	\hypo{⊢ A, Δ_{R}}
	\infer2[Cut]{Γ_{R} ⊢ Δ_{R}}
	\end{prooftree}
	\]
	\[
	\begin{prooftree}
	\hypo{Γ ⊢ A}
	\hypo{A ⊢ Δ}
	\infer2[Cut]{Γ ⊢ Δ}
	\end{prooftree}
	\]
	
	\[
	\begin{prooftree}
	\hypo{Γ_{L}, A, Γ_{R} ⊢}
	\hypo{⊢ Δ_{L}, A, Δ_{R}}
	\infer2[Cut]{Γ_{L}, Γ_{R} ⊢ Δ_{L}, Δ_{R}}
	\end{prooftree}
	\]
	
	\[
	\begin{prooftree}
	\hypo{Γ_{L}, A, Γ_{R} ⊢ Δ_{M}}
	\hypo{Γ_{M} ⊢ Δ_{L}, A, Δ_{R}}
	\infer2[Cut]{Γ_{L}, Γ_{M}, Γ_{R} ⊢ Δ_{L}, Δ_{M}, Δ_{R}}
	\end{prooftree}
	\]
\end{center}
