\section{Structural Rules}
Uses a variation of cut left.

\begin{prooftree}
\AxiomC{}
\UnaryInfC[Id]{$A \vdash A$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\AxiomC{$\Gamma_{L}, A, \Gamma_{R} \vdash C$}
\BinaryInfC[CutL]{$\Gamma_{L}, \Gamma, \Gamma_{R} \vdash C$}
\end{prooftree}

\section{Operational Rules}
$\Gamma$ non-empty in the RHS implication and negation rules.

\begin{prooftree}
\AxiomC{$\Gamma_{L}, A, B, \Gamma_{R} \vdash C$}
\UnaryInfC{$\Gamma_{L}, A \otimes B, \Gamma_{R} \vdash C$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma_{L} \vdash A$}
\AxiomC{$\Gamma_{R} \vdash B$}
\BinaryInfC{$\Gamma_{L}, \Gamma_{R} \vdash A \otimes B$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\AxiomC{$\Gamma_{L}, B, \Gamma_{R} \vdash C$}
\BinaryInfC{$\Gamma_{L}, \Gamma, A \to B, \Gamma_{R} \vdash C$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$A, \Gamma \vdash B$}
\UnaryInfC{$\Gamma \vdash A \to B$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\AxiomC{$\Gamma_{L}, B, \Gamma_{R} \vdash C$}
\BinaryInfC{$\Gamma_{L}, B \leftarrow A, \Gamma, \Gamma_{R} \vdash C$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma, A \vdash B$}
\UnaryInfC{$\Gamma \vdash B \leftarrow A$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\UnaryInfC{$\neg_{L} A, \Gamma \vdash$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\UnaryInfC{$\Gamma, \neg_{R} A \vdash$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$A, \Gamma \vdash_{L}$}
\UnaryInfC{$\Gamma \vdash_{L} \neg A$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma, A \vdash_{R}$}
\UnaryInfC{$\Gamma \vdash_{R} \neg A$}
\end{prooftree}
