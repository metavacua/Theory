\section{Operational Rules}
Uncertain about proper dualization that preserves the proper ordering for the dual.

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$ \bot \vdash $}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$C \vdash \Delta_{L}, \Delta_{R}$}
\UnaryInfC{$C \vdash \Delta_{L}, \bot , \Delta_{R}$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$0 \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{}
\UnaryInfC{$C \vdash \Delta_{L}, \top , \Delta_{R}$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$B \vdash \Delta_{L}$}
\AxiomC{$A \vdash \Delta_{R}$}
\BinaryInfC{$B \parr A \vdash \Delta_{L}, \Delta_{R}$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$C \vdash \Delta_{L}, B, A, \Delta_{L}$}
\UnaryInfC{$C \vdash \Delta_{L}, B \parr A, \Delta_{L}$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$B \vdash \Delta, A$}
\UnaryInfC{$A \nleftarrow B \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$A \vdash \Delta$}
\AxiomC{$C \vdash \Delta_{L}, B, \Delta_{R}$}
\BinaryInfC{$C \vdash \Delta_{L}, A \nleftarrow B, \Delta, \Delta_{R}$}
\end{prooftree}

\quad
\begin{prooftree}
\AxiomC{$B \vdash A, \Delta$}
\UnaryInfC{$A \nrightarrow B \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$A \vdash \Delta$}
\AxiomC{$C \vdash \Delta_{L}, B, \Delta_{R}$}
\BinaryInfC{$C \vdash \Delta_{L}, \Delta, B \nrightarrow A, \Delta_{R}$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$B \vdash \Delta$}
\AxiomC{$A \vdash \Delta$}
\BinaryInfC{$B \oplus A \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$C \vdash \Delta_{L}, A, \Delta_{R}$}
\UnaryInfC{$C \vdash \Delta_{L}, B \oplus A, \Delta_{R}$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$C \vdash \Delta_{L}, B, \Delta_{R}$}
\UnaryInfC{$C \vdash \Delta_{L}, B \oplus A, \Delta_{R}$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$A \vdash \Delta$}
\UnaryInfC{$A \land B \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$B \vdash \Delta$}
\UnaryInfC{$A \land B \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$C \vdash \Delta_{L}, B, \Delta_{R}$}
\AxiomC{$C \vdash \Delta_{L}, A, \Delta_{R}$}
\BinaryInfC{$C \vdash \Delta_{L}, B \land A, \Delta_{R}$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\vdash A, \Delta$}
\UnaryInfC{$\neg A \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\vdash \Delta, A$}
\UnaryInfC{$\neg A \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$A \vdash \Delta$}
\UnaryInfC{$\vdash \neg A, \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$A \vdash \Delta$}
\UnaryInfC{$\vdash \Delta, \neg A$}
\end{prooftree}
