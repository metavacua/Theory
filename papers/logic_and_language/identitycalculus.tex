\section{Non-Structural Identity Calculus}
\subsection{Structural Rules}

\begin{prooftree}
\AxiomC{}
\UnaryInfC[Id]{$A \vdash A$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\AxiomC{$A \vdash \Delta$}
\BinaryInfC[Cut]{$\Gamma \vdash \Delta$}
\end{prooftree}

\subsection{Operational Rules}

\subsubsection{Multiplicatives}

\begin{prooftree}
\AxiomC{$\Gamma \vdash P$}
\UnaryInfC[Substitution of Expressions]{$\Gamma \vdash P[x/E]$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma \vdash P = Q$}
\UnaryInfC[Equality of Expressions]{$\Gamma \vdash [x/P] = [x/Q]$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma \vdash P = Q$}
\AxiomC{$\Gamma \vdash Q = R$}
\BinaryInfC[Transitivity of Equality]{$\Gamma \vdash P = R$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma \vdash P$}
\AxiomC{$\Gamma \vdash P \Leftrightarrow Q$}
\BinaryInfC[Equianimity]{$\Gamma \vdash Q$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$A, B \vdash$}
\AxiomC{$\vdash A, B$}
\BinaryInfC{$A \Leftrightarrow B \vdash$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$A \vdash B$}
\AxiomC{$B \vdash A$}
\BinaryInfC{$\vdash A \Leftrightarrow B$}
\end{prooftree}
