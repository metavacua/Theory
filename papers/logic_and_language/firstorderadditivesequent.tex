Strictly visible additive order calculus with first order quantification.

\section{Structural Rules}

\begin{prooftree}
\AxiomC{}
\UnaryInfC[Id]{$A \vdash A$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\AxiomC{$A \vdash \Delta$}
\BinaryInfC[cut]{$\Gamma \vdash \Delta$}
\end{prooftree}

\section{Quantifiers}
Standard Eigenvariable Condition Quantifiers

\begin{prooftree}
\AxiomC{$P[t/x] \vdash \Delta$}
\UnaryInfC{$\forall x.P(x) \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma \vdash P[y/x]$}
\UnaryInfC{$\Gamma \vdash \forall x.P(x)$}
\end{prooftree}

in $\forall$ R the variable y must not occur free anywhere in the lower sequent.

\begin{prooftree}
\AxiomC{$P[y/x] \vdash \Delta$}
\UnaryInfC{$\exists x.P(x) \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma \vdash P[t/x]$}
\UnaryInfC{$\Gamma \vdash \exists x.P(x)$}
\end{prooftree}

in $\exists$ L the variable y must not occur free anywhere in the lower sequent.

\begin{prooftree}
\AxiomC{$P(y)\vdash$}
\UnaryInfC{$\neg \forall x.\neg P(x) \vdash$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\vdash P(t)$}
\UnaryInfC{$\vdash \neg \forall x.\neg P(x)$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$P(t)\vdash$}
\UnaryInfC{$\neg \exists x.\neg P(x) \vdash$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\vdash P(y)$}
\UnaryInfC{$\vdash \neg \exists x.\neg P(x)$}
\end{prooftree}
