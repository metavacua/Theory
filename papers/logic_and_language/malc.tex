\section{Operational Rules}
\begin{prooftree}
\hypo{$\Gamma_{L}, \Gamma_{R} \vdash C$}
\inferone{$\Gamma_{L}, 1, \Gamma_{R} \vdash C$}
\end{prooftree}
\begin{prooftree}
\inferzero{$ \vdash 1$}
\end{prooftree}
\begin{prooftree}
\inferzero{$\Gamma_{L}, 0, \Gamma_{R} \vdash C$}
\end{prooftree}
\begin{prooftree}
\inferzero{$ \Gamma \vdash \top$}
\end{prooftree}
\begin{prooftree}
\hypo{$\Gamma_{L}, A, B, \Gamma_{R} \vdash C$}
\inferone{$\Gamma_{L}, A \otimes B, \Gamma_{R} \vdash C$}
\end{prooftree}
\begin{prooftree}
\hypo{$\Gamma_{L} \vdash A$}
\hypo{$\Gamma_{R} \vdash B$}
\infertwo{$\Gamma_{L}, \Gamma_{R} \vdash A \otimes B$}
\end{prooftree}
\begin{prooftree}
\hypo{$\Gamma \vdash A$}
\hypo{$\Gamma_{L}, B, \Gamma_{R} \vdash C$}
\infertwo{$\Gamma_{L}, \Gamma, A \rightarrow B, \Gamma_{R} \vdash C$}
\end{prooftree}
\begin{prooftree}
\hypo{$A, \Gamma \vdash B$}
\inferone{$\Gamma \vdash A \rightarrow B$}
\end{prooftree}
\begin{prooftree}
\hypo{$\Gamma \vdash A$}
\hypo{$\Gamma_{L}, B, \Gamma_{R} \vdash C$}
\infertwo{$\Gamma_{L}, B \leftarrow A, \Gamma, \Gamma_{R} \vdash C$}
\end{prooftree}
\begin{prooftree}
\hypo{$\Gamma, A \vdash B$}
\inferone{$\Gamma \vdash B \leftarrow A$}
\end{prooftree}
\begin{prooftree}
\hypo{$\Gamma_{L}, A, \Gamma_{R} \vdash C$}
\hypo{$\Gamma_{L}, B, \Gamma_{R} \vdash C$}
\infertwo{$\Gamma_{L}, A \oplus B, \Gamma_{R} \vdash C$}
\end{prooftree}
\begin{prooftree}
\hypo{$\Gamma \vdash A$}
\inferone{$\Gamma \vdash A \oplus B$}
\end{prooftree}
\begin{prooftree}
\hypo{$\Gamma \vdash B$}
\inferone{$\Gamma \vdash A \oplus B$}
\end{prooftree}
\begin{prooftree}
\hypo{$\Gamma_{L}, A, \Gamma_{R} \vdash C$}
\inferone{$\Gamma_{L}, A \& B, \Gamma_{R} \vdash C$}
\end{prooftree}
\begin{prooftree}
\hypo{$\Gamma_{L}, B, \Gamma_{R} \vdash C$}
\inferone{$\Gamma_{L}, A \& B, \Gamma_{R} \vdash C$}
\end{prooftree}
\begin{prooftree}
\hypo{$\Gamma \vdash A$}
\hypo{$\Gamma \vdash B$}
\infertwo{$\Gamma \vdash A \& B$}
\end{prooftree}
\begin{prooftree}
\hypo{$ \Gamma \vdash A$}
\inferone{$ \neg_{C} A, \Gamma \vdash $}
\end{prooftree}
\begin{prooftree}
\hypo{$ \Gamma \vdash A$}
\inferone{$ \Gamma, \neg_{C} A\vdash $}
\end{prooftree}
\begin{prooftree}
\hypo{$ A, \Gamma \vdash $}
\inferone{$ \Gamma \vdash \neg_{L} A$}
\end{prooftree}
\begin{prooftree}
\hypo{$ \Gamma, A \vdash $}
\inferone{$ \Gamma \vdash \neg_{R} A$}
\end{prooftree}
