\section{Operational Rules}

\begin{prooftree}
\AxiomC{$\Gamma_L, \Gamma_R \vdash C$}
\UnaryInfC{$\Gamma_L, 1, \Gamma_R \vdash C$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\vdash 1$}
\end{prooftree}

\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\Gamma_L, 0, \Gamma_R \vdash C$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{}
\UnaryInfC{$\Gamma \vdash \top$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma_L, A, B, \Gamma_R \vdash C$}
\UnaryInfC{$\Gamma_L, A \otimes B, \Gamma_R \vdash C$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma_L \vdash A$}
\AxiomC{$\Gamma_R \vdash B$}
\BinaryInfC{$\Gamma_L, \Gamma_R \vdash A \otimes B$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\AxiomC{$\Gamma_L, B, \Gamma_R \vdash C$}
\BinaryInfC{$\Gamma_L, \Gamma, A \to B, \Gamma_R \vdash C$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$A, \Gamma \vdash B$}
\UnaryInfC{$\Gamma \vdash A \to B$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\AxiomC{$\Gamma_L, B, \Gamma_R \vdash C$}
\BinaryInfC{$\Gamma_L, B \leftarrow A, \Gamma, \Gamma_R \vdash C$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma, A \vdash B$}
\UnaryInfC{$\Gamma \vdash B \leftarrow A$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma_L, A, \Gamma_R \vdash C$}
\AxiomC{$\Gamma_L, B, \Gamma_R \vdash C$}
\BinaryInfC{$\Gamma_L, A \oplus B, \Gamma_R \vdash C$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\UnaryInfC{$\Gamma \vdash A \oplus B$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma \vdash B$}
\UnaryInfC{$\Gamma \vdash A \oplus B$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma_L, A, \Gamma_R \vdash C$}
\UnaryInfC{$\Gamma_L, A \land B, \Gamma_R \vdash C$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma_L, B, \Gamma_R \vdash C$}
\UnaryInfC{$\Gamma_L, A \land B, \Gamma_R \vdash C$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\AxiomC{$\Gamma \vdash B$}
\BinaryInfC{$\Gamma \vdash A \land B$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\UnaryInfC{$\neg_C A, \Gamma \vdash$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\UnaryInfC{$\Gamma, \neg_C A \vdash$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$A, \Gamma \vdash$}
\UnaryInfC{$\Gamma \vdash \neg_L A$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma, A \vdash$}
\UnaryInfC{$\Gamma \vdash \neg_R A$}
\end{prooftree}

\section{Structural Rules}
Uses a variation of cut left.

\begin{prooftree}
\AxiomC{}
\UnaryInfC[Id]{$A \vdash A$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\AxiomC{$\Delta, A, \Pi \vdash C$}
\BinaryInfC[CutL]{$\Delta, \Gamma, \Pi \vdash C$}
\end{prooftree}
