Strictly visible structural additive calculus with first order quantification.

\section{Structural Rules}

\begin{prooftree}
\AxiomC{}
\UnaryInfC[Id]{$A \vdash A$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\Gamma \vdash A$}
\AxiomC{$A \vdash \Delta$}
\BinaryInfC[Cut]{$\Gamma \vdash \Delta$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$\vdash C$}
\UnaryInfC[wL]{$A \vdash C$}
\end{prooftree}
\qquad
\begin{prooftree}
\AxiomC{$C \vdash$}
\UnaryInfC[Rw]{$C \vdash A$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$A, A \vdash C$}
\UnaryInfC[cL]{$A \vdash C$}
\end{prooftree}
\qquad
\begin{prooftree}
\AxiomC{$C \vdash A, A$}
\UnaryInfC[Rc]{$C \vdash A$}
\end{prooftree}

\begin{prooftree}
\AxiomC{$A, B \vdash C$}
\UnaryInfC[pL]{$B, A \vdash C$}
\end{prooftree}
\qquad
\begin{prooftree}
\AxiomC{$C \vdash A, B$}
\UnaryInfC[Rp]{$C \vdash B, A$}
\end{prooftree}

\section{Quantifiers}
Standard Eigenvariable Condition Quantifiers

\begin{prooftree}
\AxiomC{$P[t/x] \vdash \Delta$}
\UnaryInfC{$\forall x.P(x) \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma \vdash P[y/x]$}
\UnaryInfC{$\Gamma \vdash \forall x.P(x)$}
\end{prooftree}

in $\forall$ R the variable y must not occur free anywhere in the lower sequent.

\begin{prooftree}
\AxiomC{$P[y/x] \vdash \Delta$}
\UnaryInfC{$\exists x.P(x) \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\Gamma \vdash P[t/x]$}
\UnaryInfC{$\Gamma \vdash \exists x.P(x)$}
\end{prooftree}

in $\exists$ L the variable y must not occur free anywhere in the lower sequent.

\begin{prooftree}
\AxiomC{$P(y)\vdash$}
\UnaryInfC{$\neg \forall x.\neg P(x) \vdash$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\vdash P(t)$}
\UnaryInfC{$\vdash \neg \forall x.\neg P(x)$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$P(t)\vdash$}
\UnaryInfC{$\neg \exists x.\neg P(x) \vdash$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\vdash P(y)$}
\UnaryInfC{$\vdash \neg \exists x.\neg P(x)$}
\end{prooftree}


Henkin constant quantifiers

\begin{prooftree}
\AxiomC{$P(t/x) \vdash \Delta$}
\UnaryInfC{$\forall xP(x) \vdash \Delta$}
\end{prooftree}
\quad
\begin{prooftree}

\end{prooftree}
\quad
\begin{prooftree}

\end{prooftree}

\UnaryInfC{$\neg \forall x.\neg P(x) \vdash$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\vdash P(t)$}
\UnaryInfC{$\vdash \neg \forall x.\neg P(x)$}
\end{prooftree}
\quad

\AxiomC{$P(t)\vdash$}
\UnaryInfC{$\neg \exists x.\neg P(x) \vdash$}
\end{prooftree}
\quad
\begin{prooftree}
\AxiomC{$\vdash P(c_{\exists x.\neg P(x)})$}
\UnaryInfC{$\vdash \neg \exists x.\neg P(x)$}
\end{prooftree}
